Он не будет работать, пока не будет создана функция linspace(). Она создаёт список xx, необходимый для построения графика.
Готовую функцию вставьте в код и выполните его. 

import matplotlib.pyplot as plt
from math import sin, cos, sqrt

def linspace(a1, an, n):
    ...
    return [... for i in range(n)]

fig, ax = plt.subplots()
xx = linspace(-5, 5, 101)  # 101 число от -5 до 5 - значения по оси х
yy1 = [1/6 * x ** 2 - 2 for x in xx]  
# Для каждого числа х из списка хх создаётся элемент в списке уу1, имеющий значение  1/6 * x ** 2 - 2 
yy2 = [2*cos(x*2) + 1/4*cos(10*x) for x in xx]
yy3 = [sqrt(abs(x)) for x in xx]
ax.plot(xx, yy1)
ax.plot(xx, yy2)
ax.plot(xx, yy3, color = 'red')
plt.show()
